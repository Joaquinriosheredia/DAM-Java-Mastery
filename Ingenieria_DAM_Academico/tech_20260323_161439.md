```java
// Clase para implementar la lógica del algoritmo de optimización
public class OptimizadorRafaga {
    
    private Map<String, Integer> rutasPrevistas;
    
    public OptimizadorRafaga() {
        // Constructor con inicialización de datos (por ejemplo, desde base de datos)
        this.rutasPrevistas = new HashMap<>();
        
        // Ejemplo de inicialización de rutas
        rutasPrevistas.put("ruta1", 20);
        rutasPrevistas.put("ruta2", 15);
    }
    
    public String obtenerRutaOptima() {
        int minimo = Integer.MAX_VALUE;
        String rutaOptima = "";
        
        for (Map.Entry<String, Integer> entry : rutasPrevistas.entrySet()) {
            if (entry.getValue() < minimo) {
                minimo = entry.getValue();
                rutaOptima = entry.getKey();
            }
        }
        
        return rutaOptima;
    }
}
```
